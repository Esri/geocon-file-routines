173 {

177  int filetype;
179  GEOCON_BOOL flip;
180  long points_start;

184  an extent was specified when loading data.
185  */
186  int nrows;
187  int ncols;
189  double lat_min;
190  double lat_max;
191  double lon_min;
192  double lon_max;
194  /* These values are copied from the file header for convienence.
195  However, they may be changed if you want the point data written
196  in a different order.
197  */
198  int lat_dir;
199  int lon_dir;
201  /* These values are copied from the file header for convienence. */
202  double lat_delta;
203  double lon_delta;
205  double horz_scale;
206  double vert_scale;
208  /* Values for "phantom cells" around our grid. */
209  double lat_min_ghost;
210  double lat_max_ghost;
211  double lon_min_ghost;
212  double lon_max_ghost;
214  /* This will be null if data is in memory. */
215  FILE * fp;
217  /* This should be used if mutex control is needed
218  for multi-threaded access to the file when
219  transforming points and reading data on-the-fly.
220  This mutex does not need to be recursive.
221  This is currently not implemented.
222  */
223  void * mutex;
225  /* If reading data on the fly, this is null.
226  This array is always stored with points going from SW to NE.
227  */
228  GEOCON_POINT *points;
229 };
